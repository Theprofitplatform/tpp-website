/* Real-time Form Validation */
function validateField(field, rules) {
    const value = field.value.trim();
    const fieldName = field.name;
    let isValid = true;
    let message = '';

    // Email validation
    if (field.type === 'email') {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailRegex.test(value)) {
            isValid = false;
            message = 'Please enter a valid email address';
        }
    }

    // Phone validation (Australian format)
    if (field.type === 'tel') {
        const phoneRegex = /^(\+61|0)[2-9]\d{8}$/;
        const cleanPhone = value.replace(/\s|-|\(|\)/g, '');
        if (!phoneRegex.test(cleanPhone)) {
            isValid = false;
            message = 'Please enter a valid Australian phone number';
        }
    }

    // Update UI
    updateFieldValidation(field, isValid, message);
    return isValid;
}

// A/B Testing Implementation
function assignFormVariant() {
    const variants = ['control', 'value_focused', 'urgency_focused'];
    const variant = variants[Math.floor(Math.random() * variants.length)];

    sessionStorage.setItem('form_variant', variant);

    // Apply variant styles and content
    applyVariantContent(variant);

    // Track assignment
    gtag('event', 'form_variant_assigned', {
        'variant': variant,
        'page': window.location.pathname
    });

    return variant;
}

// Form Analytics Tracking
function trackFormInteraction(action, field, value) {
    gtag('event', 'form_interaction', {
        'event_category': 'Form_Optimization',
        'action': action,
        'field_name': field,
        'form_variant': sessionStorage.getItem('form_variant'),
        'step_number': getCurrentStep(),
        'time_on_page': Math.round((Date.now() - pageLoadTime) / 1000)
    });
}
