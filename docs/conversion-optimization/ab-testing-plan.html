/* A/B Testing Implementation Framework */
// Google Optimize Integration
gtag('config', 'OPT-XXXXXXX');

// Test Assignment and Tracking
function assignABTestVariant(testName, variants) {
    const userId = getUserId();
    const variantIndex = hashUserId(userId, testName) % variants.length;
    const assignedVariant = variants[variantIndex];

    // Store variant for consistency
    sessionStorage.setItem(`ab_test_${testName}`, assignedVariant);

    // Track assignment
    gtag('event', 'ab_test_assigned', {
        'test_name': testName,
        'variant': assignedVariant,
        'user_id': userId
    });

    return assignedVariant;
}

// Conversion Tracking
function trackConversion(testName, conversionType, value = null) {
    const variant = sessionStorage.getItem(`ab_test_${testName}`);

    gtag('event', 'conversion', {
        'test_name': testName,
        'variant': variant,
        'conversion_type': conversionType,
        'value': value,
        'currency': 'AUD'
    });

    // Also track to Facebook Pixel
    if (typeof fbq !== 'undefined') {
        fbq('track', 'Lead', {
            content_name: testName,
            content_category: variant,
            value: value,
            currency: 'AUD'
        });
    }
}

// Statistical Significance Calculator
function calculateSignificance(controlConversions, controlVisitors,
                              variantConversions, variantVisitors) {
    const controlRate = controlConversions / controlVisitors;
    const variantRate = variantConversions / variantVisitors;

    const pooledRate = (controlConversions + variantConversions) /
                      (controlVisitors + variantVisitors);

    const standardError = Math.sqrt(pooledRate * (1 - pooledRate) *
                         (1/controlVisitors + 1/variantVisitors));

    const zScore = (variantRate - controlRate) / standardError;
    const pValue = 2 * (1 - normalCDF(Math.abs(zScore)));

    return {
        controlRate: controlRate,
        variantRate: variantRate,
        improvement: ((variantRate - controlRate) / controlRate) * 100,
        zScore: zScore,
        pValue: pValue,
        isSignificant: pValue < 0.05
    };
}
